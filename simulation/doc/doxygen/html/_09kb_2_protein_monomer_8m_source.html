00884         %http://isoelectric.ovh.org/files/practise-isoelectric-point.html
00885         function value = get.pI(this)
00886             %retrieve
00887             if ~isempty(this.pI)
00888                 value = this.pI;
00889                 return;
00890             end
00891             
00892             %calculate
00893             numAsp = sum('D' == this.sequence);
00894             numGlu = sum('E' == this.sequence);
00895             numCys = sum('C' == this.sequence);
00896             numTyr = sum('Y' == this.sequence);
00897             numHis = sum('H' == this.sequence);
00898             numLys = sum('K' == this.sequence);
00899             numArg = sum('R' == this.sequence);
00900 
00901             pH = 6.5;             %starting point pI = 6.5 - theoretically it should be 7, but average protein pI is 6.5 so we increase the probability
00902             pHprev = 0.0;         %of finding the solution
00903             pHnext = 14.0;        %0-14 is possible pH range
00904             E = 0.01;             %epsilon means precision [pI = pH \pm E]
00905 
00906             %the infinite loop
00907             while true
00908 
00909                 % we are using pK values form Wikipedia as they give quite good approximation
00910                 % if you want you can change it
00911                 QN1 = -1/(1+10^(3.65-pH));       %C-terminal charge
00912                 QN2 = -numAsp/(1+10^(3.9-pH));   %D charge
00913                 QN3 = -numGlu/(1+10^(4.07-pH));  %E charge
00914                 QN4 = -numCys/(1+10^(8.18-pH));  %C charge
00915                 QN5 = -numTyr/(1+10^(10.46-pH)); %Y charge
00916                 QP1 = numHis/(1+10^(pH-6.04));   %H charge
00917                 QP2 = 1/(1+10^(pH-8.2));         %NH2 charge
00918                 QP3 = numLys/(1+10^(pH-10.54));  %K charge
00919                 QP4 = numArg/(1+10^(pH-12.48));  %R charge
00920 
00921                 NQ = QN1+QN2+QN3+QN4+QN5+QP1+QP2+QP3+QP4; %net charge in given pH
00922 
00923 
00924                 if pH >= 14.0
00925                     throw(MException('ProteinMonomer:error','pH higher than 14'));
00926                 end
00927 
00928                 %%%%%%%%%%%%%%%%%%%%%%%%%%   BISECTION   %%%%%%%%%%%%%%%%%%%%%%%%
00929 
00930                 %we are out of range, thus the new pH value must be smaller
00931                 if NQ<0
00932                     temp = pH;
00933                     pH = pH-((pH-pHprev)/2);
00934                     pHnext = temp;
00935 
00936                     %we used to small pH value, so we have to increase it
00937                 else
00938                     temp = pH;
00939                     pH = pH + ((pHnext-pH)/2);
00940                     pHprev = temp;
00941                 end
00942 
00943                 %terminal condition, finding isoelectric point with given precision
00944                 if (pH-pHprev<E) && (pHnext-pH<E)
00945                     break;
00946                 end
00947             end
00948 
00949             value = pH;
00950             
00951             %store
00952             this.pI = value;
00953         end
