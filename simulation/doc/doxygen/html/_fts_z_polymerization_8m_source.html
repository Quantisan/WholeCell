00002 %
00003 % @wholeCellModelID Process_FtsZPolymerization
00004 % @name             FtsZPolymerization
00005 % @description
00006 %
00007 % In the cytosol, FtsZ can exist in one of multiple states: inactivated monomer,
00008 % deactivated monomer (GDP bound), activated monomer (GTP bound), nucleated
00009 % (dimer of two activated monomers), or elongated (polymer of three or more
00010 % activated monomers). FtsZ molecules move between these states at rates
00011 % obtained from Chen et al. 2005 and Surovtsev et al. 2008.
00012 %
00013 % The maximum polymer length is a fittable parameter. We currently use 9 (40nm),
00014 % the lower end of the range given for E. coli in Anderson 2004.
00015 %
00016 % FtsZ polymerization is modeled using a set of differential equations
00017 % modified from that described in Surovtsev et al. 2008, involving the
00018 % activation, nucleation, and elongation of FtsZ polymers.
00019 % The main modifications were that the equations were simplified to not
00020 % include annealing and cyclization of FtsZ polymers.
00021 %
00022 % Solving the equations results in a real-valued distribution of monomers and
00023 % filament lengths at each time step. This process discretizes the distribution
00024 % at each time step for compatibility with the rest of the simulation.
00025 %
00026 % References
00027 % ===============
00028 % 1. Surovtsev, I.V., Morgan, J.J., Lindahl, P.A. (2008). Kinetic Modelling of the
00029 %    Assembly, Dynamic Steady State, and Contraction of the FtsZ Ring in

00031 % 2. Chen, Y., Bjornson, K., Redick, S.D., Erickson, H.P. (2005). A rapid
00032 %    fluorescence assay for ftsZ assembly indicates cooperative assembly with a

00218         %- tspan is increasing real 2 element vector with first element 0

00220         function [tout, yout] = integrateODEs(this, y0)
00221             params = [
00222                 this.activationFwd
00223                 this.activationRev
00224                 this.exchangeFwd
00225                 this.exchangeRev
00226                 this.nucleationFwd
00227                 this.nucleationRev
00228                 this.elongationFwd
00229                 this.elongationRev
00230                 this.moleculesToConcentration(this.substrates);
00231                 ];
00232 
00233             % Handle solver arguments
00234             neq = numel(y0);
00235             tfinal = this.stepSizeSec;
00236             hmax = 0.1 * this.stepSizeSec;
00237             threshold = 0.1 / edu.stanford.covert.util.ConstantUtil.nAvogadro / this.geometry.volume;
00238             rtol = 1e-2;
00239             
00240             % Initialize method parameters.
00241             pow = 1/3;
00242             d = 1 / (2 + sqrt(2));
00243             e32 = 6 + sqrt(2);
00244             
00245             % Compute the initial slope yp.
00246             t = 0;
00247             y = y0;
00248             f0 = diff(y0, params);
00249             dfdy = jacobian(y, params);
00250             
00251             % Compute an initial step size h using yp = y'(t).
00252             wt = max(abs(y), threshold);
00253             
00254             % Compute y''(t) and a better initial step size.
00255             DfDt = dfdy*f0;
00256             rh = 1.25 * sqrt(0.5 * max(abs(DfDt ./ wt))) / rtol^pow;
00257             absh = 1/rh;
00258             
00259             % Allocate memory if we're generating output.
00260             chunk = 100;
00261             tout = zeros(1, chunk);
00262             yout = zeros(neq, chunk);
00263             nout = 1;
00264             tout(nout) = t;
00265             yout(:,nout) = y;
00266             
00267             % THE MAIN LOOP
00268             done = false;
00269             notFirstStep = false;
00270             while ~done
00271                 hmin = 16*eps*t;
00272                 absh = min(hmax, max(hmin, absh));
00273                 h = absh;
00274                 
00275                 % Stretch the step if within 10% of tfinal-t.
00276                 if 1.1*absh >= abs(tfinal - t)
00277                     h = tfinal - t;
00278                     absh = abs(h);
00279                     done = true;
00280                 end
00281                 
00282                 if notFirstStep % J is already computed on first step
00283                     dfdy = jacobian(y, params);
00284                 end
00285                 
00286                 % LOOP FOR ADVANCING ONE STEP.
00287                 nofailed = true; % no failed attempts
00288                 while true % Evaluate the formula.
00289                     Miter = eye(neq) - (h*d)*dfdy;  % sparse if dfdy is sparse
00290                     k1aux = f0;
00291                     
00292                     [L,U,p] = lu(Miter,'vector');
00293                     k1 = U \ (L \ k1aux(p));
00294                     f1 = diff(y + 0.5*h*k1, params);
00295                     k2 = (U \ (L \ (f1(p) - k1(p)))) + k1;
00296                     
00297                     tnew = t + h;
00298                     if done
00299                         tnew = tfinal; % Hit end point exactly.
00300                     end
00301                     h = tnew - t; % Purify h.
00302                     
00303                     ynew = y + h*k2;
00304                     f2 = diff(ynew, params);
00305                     k3aux = (f2 - e32*(k2 - f1) - 2*(k1 - f0));
00306                     k3 = U \ (L \ k3aux(p));
00307                     
00308                     % Estimate the error.
00309                     err = (absh/6) * ...
00310                         max(abs((k1-2*k2+k3) ./ max(max(abs(y),abs(ynew)),threshold)));
00311                     
00312                     % Accept the solution only if the weighted error is no more than the
00313                     % tolerance rtol.  Estimate an h that will yield an error of rtol on
00314                     % the next step or the next try at taking this step, as the case may be,
00315                     % and use 0.8 of this value to avoid failures.
00316                     if err > rtol % Failed step
00317                         if absh <= hmin
00318                             warning('MATLAB:ode23s:IntegrationTolNotMet',['Failure at t=%e.  ' ...
00319                                 'Unable to meet integration tolerances without reducing ' ...
00320                                 'the step size below the smallest value allowed (%e) ' ...
00321                                 'at time t.'],t,hmin);
00322                             
00323                             tout = tout(1:nout);
00324                             yout = yout(:, 1:nout);
00325                             return;
00326                         end
00327                         
00328                         nofailed = false;
00329                         absh = max(hmin, absh * max(0.1, 0.8*(rtol/err)^pow));
00330                         h = absh;
00331                         done = false;
00332                     else % Successful step
00333                         break;
00334                     end
00335                 end % while true
00336                 notFirstStep = true;
00337                 
00338                 oldnout = nout;
00339                 nout = nout + 1;
00340                 if nout > length(tout)
00341                     tout = [tout, zeros(1,chunk)];  %#ok<AGROW> % requires chunk >= refine
00342                     yout = [yout, zeros(neq,chunk)]; %#ok<AGROW>
00343                 end
00344                 idx = oldnout+1:nout;
00345                 tout(idx) = tnew;
00346                 yout(:,idx) = ynew;
00347                 
00348                 if done
00349                     break;
00350                 end
00351                 
00352                 % If there were no failures compute a new h.
00353                 if nofailed
00354                     % Note that absh may shrink by 0.8, and that err may be 0.
00355                     temp = 1.25*(err/rtol)^pow;
00356                     if temp > 0.2
00357                         absh = absh / temp;
00358                     else
00359                         absh = 5.0*absh;
00360                     end
00361                 end
00362                 
00363                 % Advance the integration one step.
00364                 t = tnew;
00365                 y = ynew;
00366                 f0 = f2; % because formula is FSAL
00367                 
00368             end % while ~done
00369             
00370             tout = tout(1:nout);
00371             yout = yout(:, 1:nout);
00372         end
