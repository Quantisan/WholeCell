00658         function [tRNASequences, aaSequences] = computeTRNASequences(this, dnaSequences, findStartCodon)
00659             %options
00660             if ~exist('findStartCodon', 'var')
00661                 findStartCodon = true;
00662             end
00663 
00664             %tRNA sequence, protein lengths
00665             tRNASequences = cell(size(dnaSequences));        %tRNA sequences of each protein-coding gene
00666             aaSequences   = cell(size(dnaSequences));        %aa sequences of each protein-coding gene
00667 
00668             geneticCodeCodons          = this.geneticCodeCodons;          %DNA sequence of codons
00669             geneticCodeTRNAs           = this.geneticCodeTRNAs;           %index of tRNA within tRNAGenes corresponding to each codon
00670             geneticCodeAminoAcids      = this.geneticCodeAminoAcids;      %index of AA within tRNAGenes corresponding to each start codon
00671             geneticCodeStartCodons     = this.geneticCodeStartCodons;     %DNA sequence of start codons
00672             geneticCodeStartTRNAs      = this.geneticCodeStartTRNAs;      %index of tRNA within tRNAGenes corresponding to each start codon
00673             geneticCodeStartAminoAcids = this.geneticCodeStartAminoAcids; %index of AA within tRNAGenes corresponding to each start codon
00674 
00675             %convert nucleic acids sequences to tRNA sequences
00676             for i = 1:length(dnaSequences)
00677                 sequence = dnaSequences{i};
00678                 tRNASequence = zeros(floor(length(sequence)/3),1);
00679                 aaSequence = cell(floor(length(sequence)/3),1);
00680 
00681                 startCodonPosition = -2; %position of first base of start codon within sequence
00682                 if findStartCodon
00683                     %Start codon
00684                     %1. Step along sequence until first start codon
00685                     %2. Store tRNA corresponding to first start codon
00686                     for j=1:length(sequence)-2
00687                         tRNA=geneticCodeStartTRNAs(strcmp(geneticCodeStartCodons, sequence(j:j+2)));  %index of tRNA within tRNAGenes
00688                         if ~isempty(tRNA)
00689                             tRNASequence(1) = tRNA;
00690                             aaSequence{1} = geneticCodeStartAminoAcids{strcmp(geneticCodeStartCodons, sequence(j:j+2))};
00691                             startCodonPosition=j;
00692                             break;
00693                         end
00694                     end
00695 
00696                     if startCodonPosition == -2
00697                         throw(MException('KnowledgeBase:computeTRNASequences','No start codon found'));
00698                     end
00699                 end
00700 
00701                 %Non-start codons
00702                 %1. Step along sequence until stop codon or the end of the sequence
00703                 %2. Store tRNA corresponding to each codon
00704                 endCodonPosition = startCodonPosition; %position of first base of last protein-coding codon within sequence
00705                 for j = startCodonPosition+3:3:length(sequence)-2
00706                     tRNA = geneticCodeTRNAs(strcmp(geneticCodeCodons, sequence(j:j+2)));
00707                     if ~isempty(tRNA)
00708                         endCodonPosition=j;
00709                     else
00710                         break;
00711                     end
00712                     tRNASequence((j-startCodonPosition)/3+findStartCodon) = tRNA;
00713                     aaSequence{(j-startCodonPosition)/3+findStartCodon} = geneticCodeAminoAcids{strcmp(geneticCodeCodons, sequence(j:j+2))};
00714                 end
00715                 proteinLength = (endCodonPosition-startCodonPosition)/3+findStartCodon;
00716                 tRNASequences{i} = tRNASequence(1:proteinLength);
