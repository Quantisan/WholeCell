00156         function discoverPlots(this)
00157             this.plotNames = cell(0, 1);
00158             this.plotFunctions = cell(0, 1);
00159             
00160             metaPackage = meta.package.fromName('edu.stanford.covert.cell.sim.analysis');
00161             for i = 1:numel(metaPackage.Classes)
00162                 metaClass = metaPackage.Classes{i};
00163                 
00164                 %ignore test clases, handle clases
00165                 if any(ismember(cellfun(@(class) class.Name, metaClass.SuperClasses, 'UniformOutput', false), {'TestCase', 'handle'}))
00166                     continue;
00167                 end
00168                 
00169                 %ignore unsupported classes
00170                 className = metaClass.Name(find(metaClass.Name == '.', 1, 'last') + 1:end);
00171                 if ~ismember(className, this.plotClasses)
00172                     continue;
00173                 end
00174                 
00175                 %find plotting methods
00176                 for j = 1:numel(metaClass.Methods)
00177                     metaMethod = metaClass.Methods{j};
00178                     
00179                     %filter out methods to match required signature
00180                     if ...
00181                             length(metaMethod.Name) < 5 || ...
00182                             ~strcmp(metaMethod.Name(1:4), 'plot') || ...
00183                             ~metaMethod.Static || ...
00184                             ~strcmp(metaMethod.Access, 'public') || ...
00185                             length(metaMethod.InputNames) < 4
00186                         continue;
00187                     end
00188                     
00189                     this.plotNames = [this.plotNames; className ':' strrep(metaMethod.Name(5:end), '_', ' ')];
00190                     this.plotFunctions = [this.plotFunctions; metaClass.Name '.' metaMethod.Name];
00191                 end
00192                 
00193                 %sort plots by class, name
00194                 [~, order] = sort(this.plotNames);
00195                 this.plotNames = this.plotNames(order);
00196                 this.plotFunctions = this.plotFunctions(order);
00197             end
00198         end
