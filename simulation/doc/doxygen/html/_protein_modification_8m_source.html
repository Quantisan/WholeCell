00003 % @wholeCellModelID Process_ProteinModification
00004 % @name             Protein Modification
00005 % @description
00006 %   Biology
00007 %   ==================
00008 %   This process simulates protein modifications including
00009 %   - Adductions:
00010 %      - Ser/Thr/Tyr phosphorylation   (MG_109_DIMER)
00011 %      - lipoate ligation              (MG_270_MONOMER)
00012 %   - Ligations:
00013 %      - C-terminal glutamate ligation (MG_012_MONOMER)
00014 %
00015 %   Many protein require covalent modifications to function properly. These
00016 %   modifications are enzymatic catalyzed.
00017 %
00018 %   Knowledge Base
00019 %   ==================
00020 %   In this model of M. genitalium, there are currently 16 proteins that require
00021 %   phosphorylation at anywhere from one to 11 sites each. Only one protein
00022 %   requires lipoate ligation (pdhC), and only one requires glutamate ligation
00023 %   (rplF). All other proteins require no modification and are guaranteed to
00024 %   progress through this phase of protein maturation in a single time step.
00025 %
00026 %   The knowledge base representation of the stoichiometry of protein
00027 %   monomer modification reactions includes both the unmodified amino acid
00028 %   of the unmodified protein monomer on the left-hand-side and the
00029 %   modified amino acid of the resulting modified protein monomer on the
00030 %   right-hand-side.
00031 %
00032 %   Representation
00033 %   ==================
00034 %   The counts of unmodified and fully modified proteins are represented by the
00035 %   unmodifiedMonomers and modifiedMonomers properties. Intermediate modified
00036 %   states (proteins which have some, but not all of their requisite
00037 %   modifications) are not represented here. The molecular weights of the
00038 %   unmodified and fully modified protein mononomers are computed by the
00039 %   knowledge base protein monomer class.
00040 %
00041 %   This process uses the reactionModificationMatrix, reactionCatalysisMatrix,
00042 %   reactionStoichiometryMatrix, and enzymeBounds properties to represent the
00043 %   modifications required to mature each protein. These properties are
00044 %   initialized from the knowledge base by initializeConstants.
00045 %   reactionModificationMatrix represents the protein monomer modified by each
00046 %   reaction. reactionCatalysisMatrix represents the enzyme which catalyzes each
00047 %   reaction. enzymeBounds represents the kcat of the enzyme for each reaction.
00048 %   reactionStoichiometryMatrix represents the free metabolites reactants and
00049 %   products of each reaction. Note reactionStoichiometryMatrix used in this
00050 %   reaction is different from that of the superclass. The
00051 %   reactionStoichiometryMatrix used here does not include either the unmodified
00052 %   amino acid of the unmodified protein monomer on the left-hand-side of
00053 %   reactions or the modified amino acid of the resulting modified protein
00054 %   monomer on the right-hand-side of reactions; these amino acids are
00055 %   represented within the unmodified and modified protein monomers. For this
00056 %   process to mature a protein monomer, each of the reactions which modifies
00057 %   that protein monomer must proceed.
00058 %
00059 %   Initialization
00060 %   ==================
00061 %   All protein monomers are initialized to the mature state. This is
00062 %   accomplished by the simulation class initializeState method.
00063 %
00064 %   Simulation
00065 %   ==================
00066 %   While(true)
00067 %     1. Calculate numbers of protein monomers that can be modified based on
00068 %        substrate, enzyme, and unmodified protein monomer availability and
00069 %        kinetics.
00070 %     2. Randomly select protein monomer to modify weighted by limits
00071 %        calculated in step (1).
00072 %     3. Update substrates, enzymes, unmodified protein monomers
00073 %     4. Repeat until insufficient resources to further modify protein
00074 %        monomers
00075 %   End
00076 %

00331         function evolveState(this)
00332             %% update protein monomers that don't require modification
00333             this.modifiedMonomers(this.monomerIndexs_notmodified) = ...
00334                 this.modifiedMonomers(this.monomerIndexs_notmodified) + ...
00335                 this.unmodifiedMonomers(this.monomerIndexs_notmodified);
00336             this.unmodifiedMonomers(this.monomerIndexs_notmodified) = 0;
00337             
00338             %% stop early if no proteins require modification
00339             if ~any(this.unmodifiedMonomers)
00340                 return;
00341             end
00342             
00343             %% simulate protein monomer modification
00344             numProteins = numel(this.monomerIndexs_modified);
00345             
00346             species = [
00347                 this.substrates;
00348                 this.enzymes;
00349                 this.unmodifiedMonomers(this.monomerIndexs_modified)]';
00350             
00351             reactionFluxes = zeros(size(this.speciesReactantByproductMatrix,1),1);
00352             anyFlux = false;
00353             
00354             limits = species(ones(numProteins, 1), :) ./ this.speciesReactantMatrix;
00355             reactionLimits = min(limits, [], 2)';
00356             reactionLimits(isinf(reactionLimits) | isnan(reactionLimits) | reactionLimits < 0) = 0;
00357             isReactionInactive = reactionLimits <= 0;
00358             
00359             while true
00360                 %compute maximum number of each protein monomer species that can be modified
00361                 limits = species(ones(numProteins, 1), :) ./ max(0, this.speciesReactantByproductMatrix);
00362                 reactionLimits = min(...
00363                     this.randStream.stochasticRound(min(limits(:, this.speciesIndexs_enzymes), [], 2)), ...
00364                     min(limits(:, [1:this.speciesIndexs_enzymes(1)-1 this.speciesIndexs_enzymes(end)+1:end]), [], 2))';
00365                 reactionLimits(isReactionInactive | isinf(reactionLimits) | isnan(reactionLimits) | reactionLimits < 1) = 0;
00366                 
00367                 %stop if no more substrates can be modified
00368                 if ~any(reactionLimits)
00369                     break;
00370                 end
00371                 anyFlux = true;
00372                 
00373                 %pick reaction
00374                 selectedReaction = this.randStream.randsample(numel(reactionLimits), 1, true, reactionLimits);
00375                 reactionFluxes(selectedReaction) = reactionFluxes(selectedReaction) + 1;
00376                 
00377                 %update metabolites, enzymes, unmodified substrates
00378                 species = species - this.speciesReactantByproductMatrix(selectedReaction,:);
00379             end
00380             
00381             %stop if no reactions can proceed
00382             if ~anyFlux
00383                 return;
00384             end
00385             
00386             % update substrates
00387             this.substrates = this.substrates + ...
00388                 this.reactionStoichiometryMatrix * this.reactionModificationMatrix(:,this.monomerIndexs_modified) * reactionFluxes;
00389             
00390             %update protein monomers that require modification
00391             this.unmodifiedMonomers(this.monomerIndexs_modified) = this.unmodifiedMonomers(this.monomerIndexs_modified) - reactionFluxes;
00392             this.modifiedMonomers(this.monomerIndexs_modified)   = this.modifiedMonomers(this.monomerIndexs_modified)   + reactionFluxes;
00393         end
