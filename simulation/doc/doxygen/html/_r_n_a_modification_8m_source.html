00003 % @wholeCellModelID Process_RNAModification
00004 % @name             RNA Modification
00005 % @description
00006 %   Biology
00007 %   ==================
00008 %   This process simulates r/tRNA modifications including
00009 %   - rRNA methylation                                       MG_252_DIMER, MG_346_DIMER, MG_380_MONOMER, MG_463_MONOMER
00010 %   - rRNA pseudouridation                                   MG_209_MONOMER, MG_370_MONOMER
00011 %   - tRNA methylation                                       MG_347_DIMER, MG_445_DIMER
00012 %   - tRNA pseudouridation                                   MG_182_DIMER
00013 %   - tRNA lysidine synthetase                               MG_084_TETRAMER
00014 %   - tRNA sulfur transfer                                   MG_295_MONOMER, MG_372_MONOMER
00015 %   - tRNA uridine 5-carboxymethylaminomethyl modification   MG_008_379_TETRAMER
00016 %
00017 %   These modifications are believe to help RNAs fold properly and achieve their
00018 %   catalytically active structure. The modifications are also believed to
00019 %   improve RNA stability. In addition some tRNA modifications, namely the
00020 %   modifications near the wobble position are believed to enhance codon
00021 %   recognition. These modifications are enzymatically catalyzed by 13 proteins.
00022 %
00023 %   Knowledge Base
00024 %   ==================
00025 %   As of 8/11/2010 the M. genitalium knowledge base contained 91 RNA
00026 %   modification reactions involving 38 RNAs and 13 enzymes. Each of the 38
00027 %   RNAs is involved in 1-7 reactions. All other RNAs require no
00028 %   modification and are guaranteed to progress through this phase of RNA
00029 %   maturation in a single time step.
00030 %
00031 %   The knowledge base representation of the stoichiometry of RNA
00032 %   modification reactions includes both the unmodified nucleic acid of the
00033 %   unmodified RNA on the left-hand-side and the modified nucleic acid of
00034 %   the resulting modified RNA on the right-hand-side.
00035 %
00036 %   Representation
00037 %   ==================
00038 %   The counts of unmodified and fully modified RNAs are represented by the
00039 %   unmodifiedRNAs and modifiedRNAs properties. Intermediate modified
00040 %   states (RNAs which have some, but not all of their requisite
00041 %   modifications) are not represented here. The molecular weights of the
00042 %   unmodified and fully modified RNAs are computed by the knowledge base RNA
00043 %   classes.
00044 %
00045 %   This process uses the reactionModificationMatrix, reactionCatalysisMatrix,
00046 %   reactionStoichiometryMatrix, and enzymeBounds properties to represent the
00047 %   modifications required to mature each RNA. These properties are
00048 %   initialized from the knowledge base by initializeConstants.
00049 %   reactionModificationMatrix represents the RNA modified by each reaction.
00050 %   reactionCatalysisMatrix represents the enzyme which catalyzes each reaction.
00051 %   enzymeBounds represents the kcat of the enzyme for each reaction.
00052 %   reactionStoichiometryMatrix represents the free metabolites reactants and
00053 %   products of each reaction. Note reactionStoichiometryMatrix used in this
00054 %   reaction is different from that of the superclass. The
00055 %   reactionStoichiometryMatrix used here does not include either the unmodified
00056 %   nucleic acid of the unmodified RNA on the left-hand-side of reactions or the
00057 %   modified nucleic acid of the resulting modified RNA on the right-hand-side
00058 %   of reactions; these nucleic acids are represented within the unmodified and
00059 %   modified RNAs. For this process to mature a RNA, each of the reactions which
00060 %   modifies that RNA must proceed.
00061 %
00062 %   Initialization
00063 %   ==================
00064 %   All RNAs are initialized to the mature state. This is accomplished by the
00065 %   simulation class initializeState method.
00066 %
00067 %   Simulation
00068 %   ==================
00069 %   While(true)
00070 %     1. Calculate numbers of RNAs that can be modified based on
00071 %        substrate, enzyme, and unmodified RNA availability and kinetics.
00072 %     2. Randomly select RNA to modify weighted by limits calculated in
00073 %        step (1).
00074 %     3. Update substrates, enzymes, unmodified RNAs
00075 %     4. Repeat until insufficient resources to further modify RNAs
00076 %   End
00077 %

00297         function evolveState(this)
00298             %% update RNAs that don't require modification
00299             this.modifiedRNAs(this.rnaMask_unmodified) = ...
00300                 this.modifiedRNAs(this.rnaMask_unmodified) + ...
00301                 this.unmodifiedRNAs(this.rnaMask_unmodified);
00302             this.unmodifiedRNAs(this.rnaMask_unmodified) = 0;
00303             
00304             %% stop early if no RNAs need to be modified
00305             if ~any(this.unmodifiedRNAs)
00306                 return;
00307             end
00308 
00309             %% simulate RNA modification
00310             numRNAs = numel(this.rnaIndexs_modified);
00311             
00312             species = [
00313                 this.substrates;
00314                 this.enzymes;
00315                 this.unmodifiedRNAs(this.rnaMask_modified)]';
00316             
00317             reactionFluxes = zeros(size(this.speciesReactantByproductMatrix, 1), 1);
00318             anyFlux = false;
00319             
00320             limits = species(ones(numRNAs, 1), :) ./ this.speciesReactantMatrix;
00321             limits(:, this.substrateIndexs_water) = NaN;
00322             limits(:, this.substrateIndexs_hydrogen) = NaN;
00323             reactionLimits = min(limits, [], 2)';
00324             reactionLimits(isinf(reactionLimits) | isnan(reactionLimits) | reactionLimits < 0) = 0;
00325             isReactionInactive = reactionLimits <= 0;
00326             
00327             while true
00328                 %compute maximum number of each RNA species that can be modified
00329                 limits = species(ones(numRNAs, 1), :) ./ max(0, this.speciesReactantByproductMatrix);
00330                 limits(:, this.substrateIndexs_water) = NaN;
00331                 limits(:, this.substrateIndexs_hydrogen) = NaN;
00332                 reactionLimits = min(...
00333                     this.randStream.stochasticRound(min(limits(:, this.speciesIndexs_enzymes), [], 2)), ...
00334                     min(limits(:, [1:this.speciesIndexs_enzymes(1)-1 this.speciesIndexs_enzymes(end)+1:end]), [], 2))';
00335                 reactionLimits(isReactionInactive | isinf(reactionLimits) | isnan(reactionLimits) | reactionLimits < 1) = 0;
00336 
00337                 %stop if no more substrates can be modified
00338                 if ~any(reactionLimits); break; end;
00339                 anyFlux = true;
00340                 
00341                 %pick reaction
00342                 selectedReaction = this.randStream.randsample(numel(reactionLimits), 1, true, reactionLimits);
00343                 reactionFluxes(selectedReaction) = ...
00344                     reactionFluxes(selectedReaction) + 1;
00345 
00346                 %update metabolites, enzymes, unmodified RNAs
00347                 species = species - this.speciesReactantByproductMatrix(selectedReaction, :);
00348             end
00349 
00350             %stop if no reactions can proceed
00351             if ~anyFlux
00352                 return;
00353             end
00354             
00355             % update substrates
00356             this.substrates = this.substrates + ...
00357                 this.reactionStoichiometryMatrix * this.reactionModificationMatrix(:, this.rnaMask_modified) * reactionFluxes;
00358 
00359             %update RNAs that require modification
00360             this.unmodifiedRNAs(this.rnaMask_modified) = this.unmodifiedRNAs(this.rnaMask_modified) - reactionFluxes;
00361             this.modifiedRNAs(this.rnaMask_modified)   = this.modifiedRNAs(this.rnaMask_modified)   + reactionFluxes;
00362         end
