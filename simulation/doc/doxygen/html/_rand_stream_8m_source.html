00140         function integers = randsample(this, n, k, replacement, w)
00141             validateattributes(n, {'numeric'}, {'nonnegative', 'integer', 'finite', 'scalar'});
00142             validateattributes(k, {'numeric'}, {'nonnegative', 'integer', 'finite', 'scalar'});
00143             validateattributes(w, {'numeric'}, {'nonnegative', 'finite'});
00144             
00145             if ~replacement
00146                 validateattributes(k, {'numeric'}, {'<=', n});
00147             end
00148             if n == 0 || k <= 0 || ~any(w)
00149                 integers = zeros(0, 1);
00150                 return;
00151             end
00152             
00153             if ~replacement && k > 1
00154                 if all(w == w(1))
00155                     integers = randsample(this.randStream, n, k, replacement);
00156                     return;
00157                 end
00158                 
00159                 integers = find(w >= realmax);
00160                 if numel(integers) > k
00161                     order = this.randperm(numel(integers));
00162                     integers = integers(order(1:k));
00163                 elseif numel(integers) == k
00164                     return;
00165                 end
00166                 
00167                 w(integers) = 0;
00168                 tfs = false(size(w));
00169                 tfs(integers) = true;
00170                 nMore = k - numel(integers);
00171                 while nMore > 0 && any(w)
00172                     tmpIdxs = randsample(this.randStream, n, nMore, true, w);
00173                     tmpTfs = false(size(tmpIdxs));
00174                     for i = 1:numel(tmpIdxs)
00175                         if ~tfs(tmpIdxs(i))
00176                             tfs(tmpIdxs(i)) = true;
00177                             tmpTfs(i) = true;
00178                         end
00179                     end
00180                     w(tmpIdxs) = 0;
00181                     integers = [integers; tmpIdxs(tmpTfs)]; %#ok<AGROW>
00182                     nMore = k - numel(integers);
00183                 end
00184                 return;
00185             end
00186             
00187             if k == 1
00188                 replacement = true;
00189             end
00190             integers = randsample(this.randStream, n, k, replacement, w);

00210
