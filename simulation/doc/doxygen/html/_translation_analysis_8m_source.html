00178         function [starts, ends, pauses] = getStartsEndsPauses(sim, pos, stallPos, ribStates)
00179             rib = sim.state('Ribosome');
00180                 
00181             dpos = diff(pos, 1, 2);
00182             dStallPos = diff(stallPos, 1, 2);
00183             
00184             ends = false(size(pos));
00185             ends(:, 1:end-1) = ...
00186                 ((dpos < 0 | dStallPos < 0) & ribStates(:, 1:end-1) ~= rib.notExistValue) | ...
00187                 (ribStates(:, 1:end-1) ~= rib.notExistValue & ribStates(:, 2:end) == rib.notExistValue);
00188             ends(:, end) = ribStates(:, end) ~= rib.notExistValue;
00189             
00190             starts = false(size(pos));
00191             starts(:, 2:end) = ...
00192                 ((dpos < 0 | dStallPos < 0) & ribStates(:, 2:end) ~= rib.notExistValue) | ...
00193                 (ribStates(:, 1:end-1) == rib.notExistValue & ribStates(:, 2:end) ~= rib.notExistValue);
00194             starts(:, 1) = ribStates(:, 1) ~= rib.notExistValue;
00195             
00196             pauses = false(size(pos));
00197             pauses(dpos == 0 & ribStates(:, 1:(end - 1)) == rib.activeValue) = true;
00198         end
