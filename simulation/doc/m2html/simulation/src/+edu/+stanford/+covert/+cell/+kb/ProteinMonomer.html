0884         %http://isoelectric.ovh.org/files/practise-isoelectric-point.html
0885         function value = get.pI(this)
0886             %retrieve
0887             if ~isempty(this.pI)
0888                 value = this.pI;
0889                 return;
0890             end
0891             
0892             %calculate
0893             numAsp = sum('D' == this.sequence);
0894             numGlu = sum('E' == this.sequence);
0895             numCys = sum('C' == this.sequence);
0896             numTyr = sum('Y' == this.sequence);
0897             numHis = sum('H' == this.sequence);
0898             numLys = sum('K' == this.sequence);
0899             numArg = sum('R' == this.sequence);
0900 
0901             pH = 6.5;             %starting point pI = 6.5 - theoretically it should be 7, but average protein pI is 6.5 so we increase the probability
0902             pHprev = 0.0;         %of finding the solution
0903             pHnext = 14.0;        %0-14 is possible pH range
0904             E = 0.01;             %epsilon means precision [pI = pH \pm E]
0905 
0906             %the infinite loop
0907             while true
0908 
0909                 % we are using pK values form Wikipedia as they give quite good approximation
0910                 % if you want you can change it
0911                 QN1 = -1/(1+10^(3.65-pH));       %C-terminal charge
0912                 QN2 = -numAsp/(1+10^(3.9-pH));   %D charge
0913                 QN3 = -numGlu/(1+10^(4.07-pH));  %E charge
0914                 QN4 = -numCys/(1+10^(8.18-pH));  %C charge
0915                 QN5 = -numTyr/(1+10^(10.46-pH)); %Y charge
0916                 QP1 = numHis/(1+10^(pH-6.04));   %H charge
0917                 QP2 = 1/(1+10^(pH-8.2));         %NH2 charge
0918                 QP3 = numLys/(1+10^(pH-10.54));  %K charge
0919                 QP4 = numArg/(1+10^(pH-12.48));  %R charge
0920 
0921                 NQ = QN1+QN2+QN3+QN4+QN5+QP1+QP2+QP3+QP4; %net charge in given pH
0922 
0923 
0924                 if pH >= 14.0
0925                     throw(MException('ProteinMonomer:error','pH higher than 14'));
0926                 end
0927 
0928                 %%%%%%%%%%%%%%%%%%%%%%%%%%   BISECTION   %%%%%%%%%%%%%%%%%%%%%%%%
0929 
0930                 %we are out of range, thus the new pH value must be smaller
0931                 if NQ<0
0932                     temp = pH;
0933                     pH = pH-((pH-pHprev)/2);
0934                     pHnext = temp;
0935 
0936                     %we used to small pH value, so we have to increase it
0937                 else
0938                     temp = pH;
0939                     pH = pH + ((pHnext-pH)/2);
0940                     pHprev = temp;
0941                 end
0942 
0943                 %terminal condition, finding isoelectric point with given precision
0944                 if (pH-pHprev<E) && (pHnext-pH<E)
0945                     break;
0946                 end
0947             end
0948 
0949             value = pH;
0950             
0951             %store
0952             this.pI = value;
0953         end
